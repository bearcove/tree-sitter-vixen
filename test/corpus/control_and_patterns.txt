fn classify(flag: Bool, value: Value) -> String {
  if flag {
    "yes"
  } else if value.count > 0 {
    "maybe"
  } else {
    match value {
      Message.Hello { body } => body,
      _ => "no",
    }
  }
}

fn unpack(items: Array[String], point: Point) -> String {
  let [first, ..rest] = items;
  let { x, y } = point;
  first
}
