fn run(point: Pair, config: Config, state: State) -> U64 {
  let { x, y }: Pair = point;
  fn add_one(value: U64) -> U64 {
    value + 1
  };
  config with= {
    build.jobs: 4,
  };
  state.count = add_one x;
  return y
}
